62void init_ExtPack_LL() {
63 CPUINT_LVL1VEC = USART0_DRE_vect_num; // Set UART data register empty interrupt to higher priority as receive interrupt (Otherwise sending will be interrupted by receiving which leads to malformed command pairs)
64#if SEND_BUF_LEN > 0
65 // ------- Init ringbuffer -------
66 init_ringbuffer_metadata(send_buf, SEND_BUF_LEN, &send_buf_metadata);
67#endif
68 /*
69 * ---------- Init UART ----------
70 * UART packages: 8N1 with 1 MBAUD
71 */
72 cli();
73 //Set BAUD rate
74 USART0.BAUD = BAUD_CONST;
75 //Set 8-bit data
76 USART0.CTRLC |= USART_CHSIZE_8BIT_gc;
77 //Set RX and TX to enabled
78 USART0.CTRLB |= USART_RXEN_bm | USART_TXEN_bm;
79#if defined(__AVR_ATmega808__) || defined(__AVR_ATmega1608__)
80 PORTA_DIRSET = PIN0_bm; //Set TX to output
81 PORTA_DIRCLR = PIN1_bm; //Set RX to input
82#else
83 #error Implementation for TX and RX pin initialisation missing for this microcontroller. Add it above to fix.
84#endif
85 //Enable interrupt RX Complete
86 USART0.CTRLA |= USART_RXCIE_bm;
87 /*
88 * ---------- Init Timer ----------
89 * /8 prescaler --> 20 MHz / 8 = 2.5 MHz
90 * --> Every UART bit is 2.5 clock cycles
91 * --> 2 UART messages with 8N1 are 10 bit each --> 20 bits
92 * --> At least 50 clock cycles
93 * --> start_value = 65460 (--> 66 clock cycles (50 + 26 buffer)
94 */
95 // Normal mode is default --> No change needed
96 // No compares used --> No change needed
97 // Set prescaler to /8
98 TCA0.SINGLE.CTRLA |= TCA_SINGLE_CLKSEL_DIV8_gc | TCA_SINGLE_ENABLE_bm;
99 // Enable global interrupt
100 sei();
101}

105ext_pack_error_t send_UART_ExtPack_command(unit_t unit, uint8_t data) {
106#if SEND_BUF_LEN > 0
107 cli();
108 uint8_t is_first_command = is_buf_empty(&send_buf_metadata);
109 // Add to buffer
110 uint16_t buf_data = ((uint16_t)unit<<8) | data;
111 uint8_t ret = write_buf(&send_buf_metadata, buf_data);
112 if (is_first_command && ret == EXT_PACK_SUCCESS) {
113 // Activate data register empty interrupt
114 USART0.CTRLA |= USART_DREIE_bm;
115 }
116 sei();
117 return ret;
118#else
119 cli();
120 // Enter critical zone
121 // Send data if:
122 // - UART data register is empty
123 // - Data register empty interrupt is not active (so no data is in queue to be sent)
124 if((USART0.STATUS & USART_DREIF_bm) // Check if data register empty is set --> UART buffer empty
125 && !(USART0.CTRLA & USART_DREIE_bm)) // Check if data register empty interrupts are enabled --> buffer is reserved for following unit data
126 {
127 // UART data register empty, no data in queue to be sent and unit number valid
128 USART0.TXDATAL = unit;
129 next_data_to_send = data;
130 // Activate data register empty interrupt
131 USART0.CTRLA |= USART_DREIE_bm;
132 sei(); // Exit critical zone
133 return EXT_PACK_SUCCESS;
134 } else {
135 // Not ready to send data pair
136 sei(); // Exit critical zone
137 return EXT_PACK_FAILURE;
138 }
139#endif
140}

142/*
143 * Sends next buffer data pair or second part of data pair
144 */
145ISR(USART0_DRE_vect) {
146#if SEND_BUF_LEN > 0
147 // UART data register empty
148 if(next_data_to_send_is_buffer_pair) {
149 // Send buffer data
150 uint16_t data;
151 if(read_buf(&send_buf_metadata, &data) == EXT_PACK_SUCCESS) {
152 USART0.TXDATAL = (uint8_t)(data >> 8);
153 next_data_to_send = (uint8_t)data;
154 next_data_to_send_is_buffer_pair = 0;
155 } else {
156 // Deactivate data register empty interrupt as buffer is empty
157 USART0.CTRLA &= ~USART_DREIE_bm;
158 }
159 } else {
160 next_data_to_send_is_buffer_pair = 1;
161 // Send data part of message
162 USART0.TXDATAL = next_data_to_send;
163 // Check if command in buffer needs to be sent
164 if (is_buf_empty(&send_buf_metadata)) {
165 // Deactivate data register empty interrupt as no data in queue
166 USART0.CTRLA &= ~USART_DREIE_bm;
167 }
168 }
169#else
170 // UART data register empty
171 USART0.TXDATAL = next_data_to_send;
172 // Deactivate data register empty interrupt
173 USART0.CTRLA &= ~USART_DREIE_bm;
174#endif
175}

179/*
180 * Receives data from ExtPack via UART and triggers custom ISRs of Units.
181 * Also manages received data for units.
182 */
183ISR(USART0_RXC_vect) {
184 uint8_t errors = USART0.RXDATAH;
185 uint8_t received_data = USART0.RXDATAL;
186 if(recv_state == RECV_UNIT_NEXT_STATE) {
187 // Received unit number
188 received_unit = received_data;
189 if(errors & (USART_FERR_bm | USART_PERR_bm)) {
190 // Frame or Parity error
191 recv_state = RECV_INVALID_UNIT;
192 } else {
193 // No error
194 recv_state = RECV_DATA_NEXT_STATE;
195 }
196 // Enables reset state machine timer
197 TCA0.SINGLE.INTFLAGS |= TCA_SINGLE_OVF_bm; // Reset interrupt flags
198 TCA0.SINGLE.CNT = 65460;
199 TCA0.SINGLE.INTCTRL |= TCA_SINGLE_OVF_bm;
200 } else if(recv_state == RECV_DATA_NEXT_STATE) {
201 if(!(errors & (USART_FERR_bm | USART_PERR_bm))) {
202 // No Frame or Parity Error
203 // Valid Syntax of UART data
204 // Received unit data
205 // Disables state machine reset timer
206 TCA0.SINGLE.INTCTRL &= ~TCA_SINGLE_OVF_bm;
207 recv_state = RECV_UNIT_NEXT_STATE;
208 process_received_ExtPack_data(received_unit, received_data);
209 }
210 } else if(recv_state == RECV_INVALID_UNIT) {
211 // Received unit had an error --> ignore unit data
212 recv_state = RECV_UNIT_NEXT_STATE;
213 // Disables state machine reset timer
214 TCA0.SINGLE.INTCTRL &= ~TCA_SINGLE_OVF_bm;
215 }
216}
